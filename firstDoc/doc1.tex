\subsection{Simple example}
This is how to create the simple example you can see on this page.
\begin{verbatim}
  \pagestyle{fancy}
  \lhead{CoucouHL}
  \chead{CoucouHCenter}
  \rhead{CoucouHR}
  \lfoot{coucouFL}
  \cfoot{coucouFCenter}
  \rfoot{\thepage}
  \renewcommand{\headrulewidth}{0.4pt}
  \renewcommand{\footrulewidth}{0.4pt}
\end{verbatim}
